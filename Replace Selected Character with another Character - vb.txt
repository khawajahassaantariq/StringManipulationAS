'VisualBasic

        'Declarations
        Dim Str1 As String
        Dim Str2 As String
        Dim Char1 As Char
        Dim Char2 As Char
        Dim Char3 As Char
        Dim i As Integer

        'Initialization
        Str1 = ""
        Str2 = ""
        Char1 = ""
        Char2 = ""
        Char3 = ""
        i = 0

        'Input
        Console.Write("Enter string to process: ")
        Str1 = Console.ReadLine

        Console.Write("Enter character to find from the string: ")
        Char1 = Console.ReadLine

        Console.Write("Enter character to replace with in string: ")
        Char3 = Console.ReadLine

        'Process
        For i = 1 To Len(Str1)
            Char2 = Mid(Str1, i, 1)
            If Char1 <> Char2 Then
                Str2 = Str2 & Char2
            Else
                Str2 = Str2 & Char3
            End If
        Next i

        'Output
        Console.WriteLine("Processed string after replacement is: " & Str2)

        Console.ReadKey()
