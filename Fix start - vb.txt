
'VisualBasic

        'Declaration
        Dim s, Str3 As String
        Dim Char1, Char2 As Char
        Dim i As Integer

        'Initialization
        s = ""
        Str2 = ""
        Str3 = ""
        Char1 = ""
        Char2 = ""
        i = 0

        'Input
        Console.Write("Input String: ")
        s = Console.ReadLine

        'Process
        Str3 = Left(s, 1)
        Char1 = Left(s, 1)
        For i = 2 To Len(s)
            Char2 = Mid(s, i, 1)
            If Char1 = Char2 Then
                Str3 = Str3 & "*"
            Else : Str3 = Str3 & Char2
            End If
        Next i

        'Output
        Console.WriteLine("Your Final String is: " & Str3)
        Console.ReadKey()
