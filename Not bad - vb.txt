
'VisualBasic

        'Declaration
        Dim Str1 As String
        Dim F1 As String
        Dim F2 As String
        Dim i As Integer
        Dim F3 As Integer
        Dim F4 As Integer

        'Initialization
        Str1 = ""
        F1 = ""
        F2 = ""
        F3 = 0
        F4 = 0
        i = 0

        'Input
        Console.Write("Input String: ")
        Str1 = Console.ReadLine

        'Process
        F3 = InStr(Str1, "not")
        F1 = Left(Str1, F3 - 1)
        F4 = InStr(Str1, "bad")
        F2 = Right(Str1, Len(Str1) - F4 - 2)



        'Output
        Console.WriteLine("Your Final String is: " & F1 & "good" & F2)
        Console.ReadKey()
